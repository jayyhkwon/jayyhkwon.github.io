package BOJ._0119;

import java.util.Scanner;

public class BOJ_15988 {

    static int[] number;
    static long[] answer = new long[1000000 + 1];
    static int mod = 1000000009;

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = Integer.parseInt(sc.nextLine());
        number = new int[n];
        for (int i = 0; i < n; i++) {
            number[i] = Integer.parseInt(sc.nextLine());
        }
        solve();
        for (int i = 0; i < n; i++) {
            System.out.println((answer[number[i]]) % mod);
        }
    }

    private static void solve() {
        answer[1] = 1;
        answer[2] = 2;
        answer[3] = 4;
        for (int i = 4; i < answer.length; i++) {
            answer[i] = (answer[i - 1] % mod) + (answer[i - 2] % mod) + (answer[i - 3] % mod);
        }
    }
}
